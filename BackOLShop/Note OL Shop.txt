		*USER_ID			=	int (ai)

		*USER_TOKEN			=	varchar

2. ITEM
		*ITEM_ID			= 	int (ai)

		*ITEM_SIZE			=	varchar
		*ITEM_DESC			=	varchar
		*ITEM_STOCK			=	int
		*ITEM_PRICE			=	int
		*ITEM_WEIGHT		=	int
		
3. TRANSACTION
		*TRANSACTION_ID		=	int (ai)

		*IS_CANCELED		=	boolean
		*IS_DELIVERED		=	boolean
		*INVOICE_NUMBER		=	varchar
				
4. CLIENT
		*CLIENT_ID			=	int (ai)
		*CLIENT_NAME		=	varchar
		*CLIENT_HP			=	varchar

		*CLIENT_BANK_NAME	=	varchar

1. Format pesanan dari dalam negeri

a. 1 barang
#BELI#KODE NEGARA#NAMA PEMBELI#NAMA BANK#REKENING BANK#ALAMAT LENGKAP#KECAMATAN#PROPINSI#KODE BARANG-JUMLAH BARANG#

b. > 1 barang
#BELI#KODE NEGARA#NAMA PEMBELI#NAMA BANK#REKENING BANK#ALAMAT LENGKAP#KECAMATAN#PROPINSI#KODE BARANG-JUMLAH BARANG, KODE BARANG-JUMLAH BARANG#

2. Format pesanan dari luar negeri

a. 1 barang
#BELI#KODE NEGARA#NAMA PEMBELI#NAMA BANK#REKENING BANK#ALAMAT LENGKAP#KODE BARANG-JUMLAH BARANG#

b. > 1 barang
#BELI#KODE NEGARA#NAMA PEMBELI#NAMA BANK#REKENING BANK#ALAMAT LENGKAP#KODE BARANG-JUMLAH BARANG, KODE BARANG-JUMLAH BARANG#

3. Format cek stock barang
#CEK#KODE BARANG#UKURAN# --> akan mengeluarkan jumlah barang sesuai ukuran
(LATER!!!) #CEK#KODE BARANG# --> akan mengeluarkan jumlah barang semua ukuran
